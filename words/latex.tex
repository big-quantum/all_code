  \section{多行公式}
	% gather环境
	\subsection{gather带编号}
	\begin{gather}
		a+b=b+a \\
		f(x)=x^2+x+1
	\end{gather}
	\subsection{gather*不带编号}
	\begin{gather*}
		a+b=b+a \\
		f(x)=x^2+x+1
	\end{gather*}
	% align环境，用&进行对齐
	\subsection{align}
	\begin{align}
		a+b&=b+a \\
		f(x)&=x^2+x+1
	\end{align}
	\subsection{align*}
	\begin{align*}
		a+b&=b+a \\
		f(x)&=x^2+x+1
	\end{align*}
	% split环境，对其采用align环境对齐方式，编号在中间
	\subsection{split}
	\begin{equation}
		\begin{split}
			a+b&=b+a \\
			f(x)&=x^2+x+1
		\end{split}
	\end{equation}
	% cases环境，每行公式用&分割为两个部分，通常表示为值和后面的条件
	\subsection{cases}
	\begin{equation}
		f(x)=\begin{cases}
			1,&x>0 \\
			0,&x=0 \\
			-1,&x<0
		\end{cases}
	\end{equation}
